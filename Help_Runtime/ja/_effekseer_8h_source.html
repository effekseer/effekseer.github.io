354 inline int32_t ConvertUtf16ToUtf8(char* dst, int32_t dst_size, const char16_t* src)
355 {
356  int32_t cnt = 0;
357  const char16_t* wp = src;
358  char* cp = dst;
359 
360  if (dst_size == 0)
361  return 0;
362 
363  dst_size -= 3;
364 
365  for (cnt = 0; cnt < dst_size;)
366  {
367  char16_t wc = *wp++;
368  if (wc == 0)
369  {
370  break;
371  }
372  if ((wc & ~0x7f) == 0)
373  {
374  *cp++ = wc & 0x7f;
375  cnt += 1;
376  }
377  else if ((wc & ~0x7ff) == 0)
378  {
379  *cp++ = ((wc >> 6) & 0x1f) | 0xc0;
380  *cp++ = ((wc)&0x3f) | 0x80;
381  cnt += 2;
382  }
383  else
384  {
385  *cp++ = ((wc >> 12) & 0xf) | 0xe0;
386  *cp++ = ((wc >> 6) & 0x3f) | 0x80;
387  *cp++ = ((wc)&0x3f) | 0x80;
388  cnt += 3;
389  }
390  }
391  *cp = '\0';
392  return cnt;
393 }

402 inline int32_t ConvertUtf8ToUtf16(char16_t* dst, int32_t dst_size, const char* src)
403 {
404  int32_t i, code = 0;
405  int8_t c0, c1, c2 = 0;
406  int8_t* srci = reinterpret_cast<int8_t*>(const_cast<char*>(src));
407  if (dst_size == 0)
408  return 0;
409 
410  dst_size -= 1;
411 
412  for (i = 0; i < dst_size; i++)
413  {
414  uint16_t wc;
415 
416  c0 = *srci;
417  srci++;
418  if (c0 == '\0')
419  {
420  break;
421  }
422  // convert UTF8 to UTF16
423  code = (uint8_t)c0 >> 4;
424  if (code <= 7)
425  {
426  // 8bit character
427  wc = c0;
428  }
429  else if (code >= 12 && code <= 13)
430  {
431  // 16bit character
432  c1 = *srci;
433  srci++;
434  wc = ((c0 & 0x1F) << 6) | (c1 & 0x3F);
435  }
436  else if (code == 14)
437  {
438  // 24bit character
439  c1 = *srci;
440  srci++;
441  c2 = *srci;
442  srci++;
443  wc = ((c0 & 0x0F) << 12) | ((c1 & 0x3F) << 6) | (c2 & 0x3F);
444  }
445  else
446  {
447  continue;
448  }
449  dst[i] = wc;
450  }
451  dst[i] = 0;
452  return i;
453 }

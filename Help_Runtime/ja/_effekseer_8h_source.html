317 inline int32_t ConvertUtf16ToUtf8( int8_t* dst, int32_t dst_size, const int16_t* src )
318 {
319  int32_t cnt = 0;
320  const int16_t* wp = src;
321  int8_t* cp = dst;
322 
323  if (dst_size == 0) return 0;
324 
325  dst_size -= 3;
326 
327  for (cnt = 0; cnt < dst_size; )
328  {
329  int16_t wc = *wp++;
330  if (wc == 0)
331  {
332  break;
333  }
334  if ((wc & ~0x7f) == 0)
335  {
336  *cp++ = wc & 0x7f;
337  cnt += 1;
338  } else if ((wc & ~0x7ff) == 0)
339  {
340  *cp++ = ((wc >> 6) & 0x1f) | 0xc0;
341  *cp++ = ((wc) & 0x3f) | 0x80;
342  cnt += 2;
343  } else {
344  *cp++ = ((wc >> 12) & 0xf) | 0xe0;
345  *cp++ = ((wc >> 6) & 0x3f) | 0x80;
346  *cp++ = ((wc) & 0x3f) | 0x80;
347  cnt += 3;
348  }
349  }
350  *cp = '\0';
351  return cnt;
352 }

361 inline int32_t ConvertUtf8ToUtf16(char16_t* dst, int32_t dst_size, const char* src)
362 {
363  int32_t i, code = 0;
364  int8_t c0, c1, c2 = 0;
365  int8_t* srci = reinterpret_cast<int8_t*>(const_cast<char*>(src));
366  if (dst_size == 0)
367  return 0;
368 
369  dst_size -= 1;
370 
371  for (i = 0; i < dst_size; i++)
372  {
373  uint16_t wc;
374 
375  c0 = *srci;
376  srci++;
377  if (c0 == '\0')
378  {
379  break;
380  }
381  // convert UTF8 to UTF16
382  code = (uint8_t)c0 >> 4;
383  if (code <= 7)
384  {
385  // 8bit character
386  wc = c0;
387  }
388  else if (code >= 12 && code <= 13)
389  {
390  // 16bit character
391  c1 = *srci;
392  srci++;
393  wc = ((c0 & 0x1F) << 6) | (c1 & 0x3F);
394  }
395  else if (code == 14)
396  {
397  // 24bit character
398  c1 = *srci;
399  srci++;
400  c2 = *srci;
401  srci++;
402  wc = ((c0 & 0x0F) << 12) | ((c1 & 0x3F) << 6) | (c2 & 0x3F);
403  }
404  else
405  {
406  continue;
407  }
408  dst[i] = wc;
409  }
410  dst[i] = 0;
411  return i;
412 }

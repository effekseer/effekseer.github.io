267 inline int32_t ConvertUtf16ToUtf8( int8_t* dst, int32_t dst_size, const int16_t* src )
268 {
269  int32_t cnt = 0;
270  const int16_t* wp = src;
271  int8_t* cp = dst;
272 
273  if (dst_size == 0) return 0;
274 
275  dst_size -= 3;
276 
277  for (cnt = 0; cnt < dst_size; )
278  {
279  int16_t wc = *wp++;
280  if (wc == 0)
281  {
282  break;
283  }
284  if ((wc & ~0x7f) == 0)
285  {
286  *cp++ = wc & 0x7f;
287  cnt += 1;
288  } else if ((wc & ~0x7ff) == 0)
289  {
290  *cp++ = ((wc >> 6) & 0x1f) | 0xc0;
291  *cp++ = ((wc) & 0x3f) | 0x80;
292  cnt += 2;
293  } else {
294  *cp++ = ((wc >> 12) & 0xf) | 0xe0;
295  *cp++ = ((wc >> 6) & 0x3f) | 0x80;
296  *cp++ = ((wc) & 0x3f) | 0x80;
297  cnt += 3;
298  }
299  }
300  *cp = '\0';
301  return cnt;
302 }

311 inline int32_t ConvertUtf8ToUtf16( int16_t* dst, int32_t dst_size, const int8_t* src )
312 {
313  int32_t i, code;
314  int8_t c0, c1, c2;
315 
316  if (dst_size == 0) return 0;
317 
318  dst_size -= 1;
319 
320  for (i = 0; i < dst_size; i++)
321  {
322  int16_t wc;
323 
324  c0 = *src++;
325  if (c0 == '\0')
326  {
327  break;
328  }
329  // UTF8からUTF16に変換
330  code = (uint8_t)c0 >> 4;
331  if (code <= 7)
332  {
333  // 8bit文字
334  wc = c0;
335  }
336  else if (code >= 12 && code <= 13)
337  {
338  // 16bit文字
339  c1 = *src++;
340  wc = ((c0 & 0x1F) << 6) | (c1 & 0x3F);
341  }
342  else if (code == 14)
343  {
344  // 24bit文字
345  c1 = *src++;
346  c2 = *src++;
347  wc = ((c0 & 0x0F) << 12) | ((c1 & 0x3F) << 6) | (c2 & 0x3F);
348  }
349  else
350  {
351  continue;
352  }
353  dst[i] = wc;
354  }
355  dst[i] = 0;
356  return i;
357 }

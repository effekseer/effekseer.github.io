265 inline int32_t ConvertUtf16ToUtf8( int8_t* dst, int32_t dst_size, const int16_t* src )
266 {
267  int32_t cnt = 0;
268  const int16_t* wp = src;
269  int8_t* cp = dst;
270 
271  if (dst_size == 0) return 0;
272 
273  dst_size -= 3;
274 
275  for (cnt = 0; cnt < dst_size; )
276  {
277  int16_t wc = *wp++;
278  if (wc == 0)
279  {
280  break;
281  }
282  if ((wc & ~0x7f) == 0)
283  {
284  *cp++ = wc & 0x7f;
285  cnt += 1;
286  } else if ((wc & ~0x7ff) == 0)
287  {
288  *cp++ = ((wc >> 6) & 0x1f) | 0xc0;
289  *cp++ = ((wc) & 0x3f) | 0x80;
290  cnt += 2;
291  } else {
292  *cp++ = ((wc >> 12) & 0xf) | 0xe0;
293  *cp++ = ((wc >> 6) & 0x3f) | 0x80;
294  *cp++ = ((wc) & 0x3f) | 0x80;
295  cnt += 3;
296  }
297  }
298  *cp = '\0';
299  return cnt;
300 }

309 inline int32_t ConvertUtf8ToUtf16( int16_t* dst, int32_t dst_size, const int8_t* src )
310 {
311  int32_t i, code;
312  int8_t c0, c1, c2;
313 
314  if (dst_size == 0) return 0;
315 
316  dst_size -= 1;
317 
318  for (i = 0; i < dst_size; i++)
319  {
320  int16_t wc;
321 
322  c0 = *src++;
323  if (c0 == '\0')
324  {
325  break;
326  }
327  // UTF8からUTF16に変換
328  code = (uint8_t)c0 >> 4;
329  if (code <= 7)
330  {
331  // 8bit文字
332  wc = c0;
333  }
334  else if (code >= 12 && code <= 13)
335  {
336  // 16bit文字
337  c1 = *src++;
338  wc = ((c0 & 0x1F) << 6) | (c1 & 0x3F);
339  }
340  else if (code == 14)
341  {
342  // 24bit文字
343  c1 = *src++;
344  c2 = *src++;
345  wc = ((c0 & 0x0F) << 12) | ((c1 & 0x3F) << 6) | (c2 & 0x3F);
346  }
347  else
348  {
349  continue;
350  }
351  dst[i] = wc;
352  }
353  dst[i] = 0;
354  return i;
355 }
